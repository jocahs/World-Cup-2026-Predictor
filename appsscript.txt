function doGet(e) {
  const action = e.parameter.action;
  
  try {
    const sheet = SpreadsheetApp.getActiveSpreadsheet();
    let picksSheet = sheet.getSheetByName("Picks");
    let matchesSheet = sheet.getSheetByName("Matches");
    
    if (!picksSheet) {
      picksSheet = sheet.insertSheet("Picks");
      picksSheet.appendRow(["Timestamp", "Player", "Match", "Pick"]);
    }
    
    if (!matchesSheet) {
      matchesSheet = sheet.insertSheet("Matches");
      matchesSheet.appendRow(["Match", "Group", "Date", "Time", "Home Team", "Visitor Team", "Result", "Home Score", "Visitor Score", "Stage", "Transmission"]);
    }
    
    // LOAD DATA
    if (action === "load") {
      const matchesData = matchesSheet.getDataRange().getValues();
      const matches = [];
      const results = {};
      
      for (let i = 1; i < matchesData.length; i++) {
        const matchName = matchesData[i][0];
        const group = matchesData[i][1];
        let rawDate = matchesData[i][2];
        const time = matchesData[i][3];
        const homeTeam = matchesData[i][4];
        const visitorTeam = matchesData[i][5];
        const result = matchesData[i][6];
        const homeScore = matchesData[i][7];
        const visitorScore = matchesData[i][8];
        const stage = matchesData[i][9] || "group";
        const transmission = matchesData[i][10] || "";
        
        let dateStr = "";
        if (rawDate instanceof Date) {
          dateStr = Utilities.formatDate(rawDate, Session.getScriptTimeZone(), "yyyy/MM/dd");
        } else if (typeof rawDate === 'string') {
          dateStr = rawDate;
        } else {
          dateStr = "";
        }
        
        matches.push({
          name: matchName,
          group: group,
          date: dateStr,
          time: time,
          homeTeam: homeTeam,
          visitorTeam: visitorTeam,
          stage: stage,
          transmission: transmission
        });
        
        if (result) {
          results[matchName] = {
            result: result.toString().replace(/^'/, ''),
            score: homeScore + "-" + visitorScore
          };
        }
      }
      
      const picks = [];
      const picksData = picksSheet.getDataRange().getValues();
      for (let i = 1; i < picksData.length; i++) {
        picks.push({
          timestamp: picksData[i][0],
          player: picksData[i][1],
          match: picksData[i][2],
          pick: picksData[i][3]
        });
      }
      
      return ContentService.createTextOutput(JSON.stringify({
        success: true,
        picks: picks,
        results: results,
        matches: matches
      })).setMimeType(ContentService.MimeType.JSON);
    }
    
    // SAVE PICKS
    if (action === "save") {
      const player = e.parameter.player;
      const picks = JSON.parse(e.parameter.picks);
      for (let i = 0; i < picks.length; i++) {
        picksSheet.appendRow([new Date(), player, picks[i].match, picks[i].pick]);
      }
      return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON);
    }
    
    // SAVE SCORES
    if (action === "saveScores") {
      const scoresData = JSON.parse(e.parameter.data);
      const matchesData = matchesSheet.getDataRange().getValues();
      
      for (let i = 1; i < matchesData.length; i++) {
        const matchName = matchesData[i][0];
        const update = scoresData.find(s => s.match === matchName);
        
        if (update) {
          matchesSheet.getRange(i + 1, 7).setValue(update.result);
          matchesSheet.getRange(i + 1, 8).setValue(update.homeScore);
          matchesSheet.getRange(i + 1, 9).setValue(update.awayScore);
        }
      }
      
      return ContentService.createTextOutput(JSON.stringify({success: true, message: "Scores updated"})).setMimeType(ContentService.MimeType.JSON);
    }

    // UPDATE SINGLE PICK
    if (action === "updatePick") {
      const player = e.parameter.player;
      const match = e.parameter.match;
      const pick = e.parameter.pick;
      
      const sheet = SpreadsheetApp.getActiveSpreadsheet();
      let picksSheet = sheet.getSheetByName("Picks");
      
      if (!picksSheet) {
        picksSheet = sheet.insertSheet("Picks");
        picksSheet.appendRow(["Timestamp", "Player", "Match", "Pick"]);
      }
      
      const currentData = picksSheet.getDataRange().getValues();
      let found = false;
      
      // Find existing pick and update it
      for (let i = 1; i < currentData.length; i++) {
        if (currentData[i][1] === player && currentData[i][2] === match) {
          picksSheet.getRange(i + 1, 4).setValue(pick); // Update pick only
          found = true;
          break;
        }
      }
      
      // If not found, add new row
      if (!found) {
        picksSheet.appendRow([new Date(), player, match, pick]);
      }
      
      return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON);
    }

    
    return ContentService.createTextOutput(JSON.stringify({success: false, error: "Invalid action: " + action})).setMimeType(ContentService.MimeType.JSON);
    
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}

function doPost(e) {
  try {
    const data = JSON.parse(e.postData.contents);
    const action = data.action;
    
    const sheet = SpreadsheetApp.getActiveSpreadsheet();
    let picksSheet = sheet.getSheetByName("Picks");
    
    if (!picksSheet) {
      picksSheet = sheet.insertSheet("Picks");
      picksSheet.appendRow(["Timestamp", "Player", "Match", "Pick", "Group", "Date", "Time"]);
    }
    
    if (action === "save") {
      const player = data.player;
      const picks = data.picks;
      
      // Save all picks
      for (let i = 0; i < picks.length; i++) {
        picksSheet.appendRow([
          new Date(), 
          player, 
          picks[i].match, 
          picks[i].pick,
          picks[i].group || "",
          picks[i].date || "",
          picks[i].time || ""
        ]);
      }
      
      return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON);
    }
    
    if (action === "updatePicks") {
      const player = data.player;
      const updates = data.updates;
      const removals = data.removals;
      
      const currentData = picksSheet.getDataRange().getValues();
      
      // Remove picks
      for (let i = currentData.length - 1; i >= 1; i--) {
        if (currentData[i][1] === player && removals.includes(currentData[i][2])) {
          picksSheet.deleteRow(i + 1);
        }
      }
      
      // Update or add picks
      for (let update of updates) {
        let found = false;
        const newData = picksSheet.getDataRange().getValues();
        for (let i = 1; i < newData.length; i++) {
          if (newData[i][1] === player && newData[i][2] === update.match) {
            picksSheet.getRange(i + 1, 4).setValue(update.pick);
            if (update.group) picksSheet.getRange(i + 1, 5).setValue(update.group);
            if (update.date) picksSheet.getRange(i + 1, 6).setValue(update.date);
            if (update.time) picksSheet.getRange(i + 1, 7).setValue(update.time);
            found = true;
            break;
          }
        }
        if (!found) {
          picksSheet.appendRow([
            new Date(), player, update.match, update.pick,
            update.group || "", update.date || "", update.time || ""
          ]);
        }
      }
      
      return ContentService.createTextOutput(JSON.stringify({success: true})).setMimeType(ContentService.MimeType.JSON);
    }
    
    return ContentService.createTextOutput(JSON.stringify({success: false, error: "Invalid action"})).setMimeType(ContentService.MimeType.JSON);
    
  } catch(error) {
    return ContentService.createTextOutput(JSON.stringify({success: false, error: error.toString()})).setMimeType(ContentService.MimeType.JSON);
  }
}
